def conv2d_maxpool(x_tensor, conv_num_outputs, conv_ksize, conv_strides, pool_ksize, pool_strides):
    """
    Apply convolution then max pooling to x_tensor
    :param x_tensor: TensorFlow Tensor
    :param conv_num_outputs: Number of outputs for the convolutional layer
    :param conv_ksize: kernal size 2-D Tuple for the convolutional layer
    :param conv_strides: Stride 2-D Tuple for convolution
    :param pool_ksize: kernal size 2-D Tuple for pool
    :param pool_strides: Stride 2-D Tuple for pool
    : return: A tensor that represents convolution and max pooling of x_tensor
    """
    # TODO: Implement Function
    # Weight and bias
    weight_shape = (*conv_ksize, x_tensor.get_shape().as_list()[3], conv_num_outputs)
    initialize = 5e-2 #np.sqrt(2.0/np.prod(weight_shape[:2])) # weight initializer most suitable for relus [He et al]
    
    weight = tf.Variable(tf.truncated_normal(weight_shape, mean=0, stddev=initialize, dtype=tf.float32))
    bias = tf.Variable(tf.zeros(conv_num_outputs))
    
    # Apply Convolution
    conv_layer = tf.nn.conv2d(x_tensor, weight, strides=[1, *conv_strides, 1], padding='SAME')
    
    # Add bias
    conv_layer = tf.nn.bias_add(conv_layer, bias)
    
    # Apply activation function
    conv_layer = tf.nn.relu(conv_layer)
    #print (conv_layer.get_shape())

    # Apply Max Pooling
    conv_layer = tf.nn.max_pool(conv_layer, ksize=[1, *pool_ksize, 1], strides=[1, *pool_strides, 1], padding='SAME')
    #print (conv_layer.get_shape())

    return conv_layer

def fully_conn(x_tensor, num_outputs):
    """
    Apply a fully connected layer to x_tensor using weight and bias
    : x_tensor: A 2-D tensor where the first dimension is batch size.
    : num_outputs: The number of output that the new tensor should be.
    : return: A 2-D tensor where the second dimension is num_outputs.
    """
    # TODO: Implement Function
    num_input = x_tensor.get_shape().as_list()[1]

    initialize = 5e-2 #np.sqrt(2.0/num_input)

    weight = tf.Variable(tf.truncated_normal((num_input, num_outputs), mean=0, stddev=initialize, dtype=tf.float32))
    bias = tf.Variable(tf.zeros(num_outputs))
    
    # Logits - xW + b
    logits = tf.add(tf.matmul(x_tensor, weight), bias)
    
    return tf.nn.relu(logits)

def conv_net(x, keep_prob):
    """
    Create a convolutional neural network model
    : x: Placeholder tensor that holds image data.
    : keep_prob: Placeholder tensor that hold dropout keep probability.
    : return: Tensor that represents logits
    """
    con_tensor = x

    # TODO: Apply 1, 2, or 3 Convolution and Max Pool layers
    #    Play around with different number of outputs, kernel size and stride
    # Function Definition from Above:
    #    conv2d_maxpool(x_tensor, conv_num_outputs, conv_ksize, conv_strides, pool_ksize, pool_strides)

    # parameters remain the same for the different convolution networks applied below
    # as defined in VGGNet by [Simonyan and Zisserman, 2014]
    
    ksize = 3   #kernel size of (3,3)
    stride = 1  #stride for kernel size of (1, 1) -- the dimensions of the input remains the same
    pksize = 2  #pool kernel size of (2, 2) 
    pstride = 2 #pool stride size of (2, 2) -- reducing the dimension of the input by half
    
    # after application of each maximum pooling layer, a drop out layer is added for regularization
    # each of the layers has output layers in the powers of 2 for better performance
    # the depth of the output layer are doubled as the dimension of the input is halved
    # pool kernel size and stride of 1 for maximum pooling to act as identity
    con_tensor = conv2d_maxpool(con_tensor, 64, (ksize, ksize), (stride, stride) , (1, 1), (1, 1))
    con_tensor = conv2d_maxpool(con_tensor, 64, (ksize, ksize), (stride, stride) , (pksize, pksize), (pstride, pstride))
    con_tensor = tf.nn.dropout(con_tensor, keep_prob) #drop out to regularize

    con_tensor = conv2d_maxpool(con_tensor, 128, (ksize, ksize), (stride, stride) , (1, 1), (1, 1))
    con_tensor = conv2d_maxpool(con_tensor, 128, (ksize, ksize), (stride, stride) , (pksize, pksize), (pstride, pstride))
    con_tensor = tf.nn.dropout(con_tensor, keep_prob) #drop out to regularize

    con_tensor = conv2d_maxpool(con_tensor, 256, (ksize, ksize), (stride, stride) , (1, 1), (1, 1))
    con_tensor = conv2d_maxpool(con_tensor, 256, (ksize, ksize), (stride, stride) , (pksize, pksize), (pstride, pstride))
    con_tensor = tf.nn.dropout(con_tensor, keep_prob) #drop out to regularize
    
    # TODO: Apply a Flatten Layer
    # Function Definition from Above:
    #   flatten(x_tensor)
    con_tensor = flatten(con_tensor)
    
    # TODO: Apply 1, 2, or 3 Fully Connected Layers
    #    Play around with different number of outputs
    # Function Definition from Above:
    #   fully_conn(x_tensor, num_outputs)
    
    # 2 fully connected layers are added with output volumn of power of 2
    # drop out layer is added after every fully connected layer for regularization
    con_tensor = fully_conn(con_tensor, 512)
    
    # doubling the keep_prob so that it does not drop out extensive data in case of fully connected. 
    # another approach would be to not to use drop out in this layer and then add it unchanged in the layer
    # below, but intitutively I felt that adding it here but doubling it if less than 0.5 would allow me 
    # to regularize both the fully connected layers.
    prob = tf.cond(keep_prob < 0.5, lambda: tf.multiply(keep_prob, tf.constant(2.0)), lambda: keep_prob) 
    con_tensor = tf.nn.dropout(con_tensor, prob) #drop out to regularize - using double the keep_prob

    con_tensor = fully_conn(con_tensor, 256)
    con_tensor = tf.nn.dropout(con_tensor, prob) #drop out to regularize - using double the keep_prob

    # TODO: Apply an Output Layer
    #    Set this to the number of classes
    # Function Definition from Above:
    #   output(x_tensor, num_outputs)
    con_tensor = output(con_tensor, 10) # output layer is applied

Hyperparameters¶

Tune the following parameters:

  • Set epochs to the number of iterations until the network stops learning or start overfitting
  • Set batch_size to the highest number that your machine has memory for. Most people set them to common sizes of memory:
    • 64
    • 128
    • 256
    • ...
  • Set keep_probability to the probability of keeping a node using dropout

import tensorflow as tf
import pickle
import helper
import random

# Set batch size if not already set
try:
    if batch_size:
        pass
except NameError:
    batch_size = 64

save_model_path = './image_classification'
n_samples = 4
top_n_predictions = 3

def test_model():
    """
    Test the saved model against the test dataset
    """

    test_features, test_labels = pickle.load(open('preprocess_training.p', mode='rb'))
    loaded_graph = tf.Graph()

    with tf.Session(graph=loaded_graph) as sess:
        # Load model
        loader = tf.train.import_meta_graph(save_model_path + '.meta')
        loader.restore(sess, save_model_path)

        # Get Tensors from loaded model
        loaded_x = loaded_graph.get_tensor_by_name('x:0')
        loaded_y = loaded_graph.get_tensor_by_name('y:0')
        loaded_keep_prob = loaded_graph.get_tensor_by_name('keep_prob:0')
        loaded_logits = loaded_graph.get_tensor_by_name('logits:0')
        loaded_acc = loaded_graph.get_tensor_by_name('accuracy:0')
        
        # Get accuracy in batches for memory limitations
        test_batch_acc_total = 0
        test_batch_count = 0
        
        for train_feature_batch, train_label_batch in helper.batch_features_labels(test_features, test_labels, batch_size):
            test_batch_acc_total += sess.run(
                loaded_acc,
                feed_dict={loaded_x: train_feature_batch, loaded_y: train_label_batch, loaded_keep_prob: 1.0})
            test_batch_count += 1

        print('Testing Accuracy: {}\n'.format(test_batch_acc_total/test_batch_count))

        # Print Random Samples
        random_test_features, random_test_labels = tuple(zip(*random.sample(list(zip(test_features, test_labels)), n_samples)))
        random_test_predictions = sess.run(
            tf.nn.top_k(tf.nn.softmax(loaded_logits), top_n_predictions),
            feed_dict={loaded_x: random_test_features, loaded_y: random_test_labels, loaded_keep_prob: 1.0})
        helper.display_image_predictions(random_test_features, random_test_labels, random_test_predictions)
